Events

Bind a named function:


function clickHandler() {
  console.log('clicked!');
}
$('#button').on('click', clickHandler);

Or bind an anonymous function:


$('#button').on('click', function() {
  console.log('clicked!');
});

The Event Object

The 'event' parameter provides a return value that is treated in a special way, telling us a variety of things about our event, such as the type or target element.


$('a').on('click keypress', function(event) {
  console.log(event);
  
  if(event.type === 'keypress'){
    console.log(event.keyCode);
  }
});

Events

Preventing default actions


  $('form').on('submit', function(event) {
    event.preventDefault();

    console.log('Not submitting the form!');
    // Maybe I want to do stuff first.
  });

  var _this = this;
  console.log(_this);

  $('a').on('click', function(event) {
    console.log(this);
    console.log(_this);

Chaining


var banner = $('#banner');
banner.css('color', 'red');
banner.html('Welcome!');
banner.show();

Is the same as:


banner.css('color', 'red').html('Welcome!').show();

Is the same as:


banner.css('color', 'red')
 .html('Welcome!')
 .show();

Promises

Promises are sort-of like events, providing callbacks for processes yet to be completed. But they help us deal with JS being single-threaded.


$( "button" ).on( "click", function() {
  $( "p" ).append( "Started..." );

  $( "div" ).each(function( i ) {
    $( this ).fadeIn().fadeOut( 1000 * ( i + 1 ) );
  });

  $( "div" ).promise().done(function() {
    $( "p" ).append( " Finished! " );
  });
});

Best Practices

Put selectors in variables for reuse

This:


var elPageHeader = $('#header'),
    elHeaderListItems = elPageHeader.find('li'),
    elHeaderLinks = elPageHeader.find('a');

elHeaderListItems.addClass('fancy');
elHeaderLinks.attr('title', 'The best links');

Instead of this:


$('#header li').addClass('fancy');
$('#header li a').attr('title', 'Some cool link');
